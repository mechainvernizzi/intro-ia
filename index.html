async function sbFetch(table) { try { const r = await fetch(`${SB_URL}/rest/v1/${table}?select=*&limit=500&order=id`, { headers: { apikey: SB_KEY, Authorization: `Bearer ${SB_KEY}` }, cache: 'no-store' }); if (!r.ok) { const errorText = await r.text(); console.error('Supabase error:', { table, status: r.status, statusText: r.statusText, error: errorText }); return null; } const d = await r.json(); console.log(`Supabase loaded: ${table}`, d); return Array.isArray(d) && d.length ? d : null; } catch(e) { console.error('sbFetch failed:', table, e); return null; } }
